set size 0.55,0.9

set xtics 0,30
set mytics 10
set mxtics 3
set xlabel "{/Symbol q}_{LAB} [deg]"
set ylabel "^{209}Bi d{/Symbol s}/d{/Symbol W} [mb/sr]"
set xrange [ 0 : 180 ] 
set yrange [ 0.001 : 1000 ]

set label "8.03(x10)"         at 50,2.5
